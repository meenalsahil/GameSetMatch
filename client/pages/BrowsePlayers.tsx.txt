// client/pages/BrowsePlayers.tsx
import { useQuery } from "@tanstack/react-query";
import { Link } from "wouter";

import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import { Skeleton } from "@/components/ui/skeleton";
import Footer from "@/components/Footer";

import {
  MapPin,
  Trophy,
  User,
  ArrowRight,
  ShieldCheck,
} from "lucide-react";

type Player = {
  id: number | string;
  fullName: string;
  location?: string | null;
  ranking?: number | string | null;
  specialization?: string | null;
  bio?: string | null;
  fundingGoals?: string | null;
  videoUrl?: string | null;
  photoUrl?: string | null;
  country?: string | null;
  age?: number | null;
  atpVerified?: boolean | null;
  atpVerificationScore?: number | null;
};

export default function BrowsePlayers() {
  const {
    data: players,
    isLoading,
    isError,
  } = useQuery<Player[]>({
    queryKey: ["/api/players"],
    queryFn: async () => {
      const res = await fetch("/api/players");
      if (!res.ok) throw new Error("Failed to load players");
      return res.json();
    },
  });

  return (
    <div className="min-h-screen flex flex-col bg-background">
      {/* Hero / Header */}
      <header className="border-b bg-gradient-to-b from-primary/10 to-background">
        <div className="container mx-auto px-6 py-10 flex flex-col gap-4">
          <div className="flex flex-col md:flex-row md:items-center md:justify-between gap-4">
            <div>
              <h1 className="text-3xl md:text-4xl font-bold tracking-tight">
                Browse Players
              </h1>
              <p className="text-muted-foreground mt-1 max-w-xl">
                Discover hardworking junior & college tennis players and help
                them cover travel, coaching, and tournament costs.
              </p>
            </div>

            {/* Simple nav actions (does NOT replace your main nav, just extra) */}
            <div className="flex flex-wrap gap-2 justify-start md:justify-end">
              <Button variant="outline" asChild>
                <Link href="/">Home</Link>
              </Button>
              <Button asChild>
                <Link href="/signup/player">Get Started as a Player</Link>
              </Button>
            </div>
          </div>
        </div>
      </header>

      {/* Main content */}
      <main className="flex-1">
        <div className="container mx-auto px-6 py-10">
          {isLoading && (
            <div className="grid md:grid-cols-3 gap-6">
              {Array.from({ length: 6 }).map((_, i) => (
                <Card key={i}>
                  <CardHeader>
                    <div className="flex items-center gap-3">
                      <Skeleton className="h-12 w-12 rounded-full" />
                      <div className="space-y-2 w-full">
                        <Skeleton className="h-4 w-1/2" />
                        <Skeleton className="h-3 w-1/3" />
                      </div>
                    </div>
                  </CardHeader>
                  <CardContent className="space-y-3">
                    <Skeleton className="h-3 w-full" />
                    <Skeleton className="h-3 w-5/6" />
                    <Skeleton className="h-8 w-32" />
                  </CardContent>
                </Card>
              ))}
            </div>
          )}

          {isError && (
            <div className="text-center text-destructive">
              Failed to load players. Please refresh the page.
            </div>
          )}

          {!isLoading && !isError && (!players || players.length === 0) && (
            <div className="text-center py-16 space-y-4">
              <h2 className="text-2xl font-semibold">
                No players are visible yet.
              </h2>
              <p className="text-muted-foreground max-w-md mx-auto">
                Once players have created their profiles and been approved,
                they will appear here so sponsors can support them.
              </p>
              <Button asChild>
                <Link href="/signup/player">
                  <ArrowRight className="h-4 w-4 mr-2" />
                  Be the first player to join
                </Link>
              </Button>
            </div>
          )}

          {!isLoading && !isError && players && players.length > 0 && (
            <div className="grid md:grid-cols-3 gap-6">
              {players.map((player) => (
                <Card key={player.id} className="flex flex-col">
                  <CardHeader>
                    <div className="flex items-center gap-3">
                      <div className="h-12 w-12 rounded-full bg-muted flex items-center justify-center overflow-hidden">
                        {player.photoUrl ? (
                          // eslint-disable-next-line @next/next/no-img-element
                          <img
                            src={player.photoUrl}
                            alt={player.fullName}
                            className="h-full w-full object-cover"
                          />
                        ) : (
                          <User className="h-6 w-6 text-muted-foreground" />
                        )}
                      </div>
                      <div>
                        <CardTitle className="text-lg">
                          {player.fullName}
                        </CardTitle>
                        <div className="flex flex-wrap items-center gap-2 text-xs text-muted-foreground">
                          {player.location && (
                            <span className="flex items-center gap-1">
                              <MapPin className="h-3 w-3" />
                              {player.location}
                            </span>
                          )}
                          {player.ranking && (
                            <span className="flex items-center gap-1">
                              <Trophy className="h-3 w-3" />
                              Rank: {player.ranking}
                            </span>
                          )}
                        </div>
                      </div>
                    </div>
                  </CardHeader>

                  <CardContent className="flex-1 flex flex-col gap-3">
                    {player.specialization && (
                      <Badge variant="outline" className="w-fit">
                        {player.specialization}
                      </Badge>
                    )}

                    {player.bio && (
                      <p className="text-sm text-muted-foreground line-clamp-3">
                        {player.bio}
                      </p>
                    )}

                    {player.fundingGoals && (
                      <p className="text-xs text-muted-foreground line-clamp-2">
                        <span className="font-medium">Funding goals: </span>
                        {player.fundingGoals}
                      </p>
                    )}

                    <div className="flex items-center justify-between pt-2 mt-auto">
                      <Button size="sm" asChild>
                        <Link href={`/players/${player.id}`}>
                          Support Player
                          <ArrowRight className="h-4 w-4 ml-2" />
                        </Link>
                      </Button>

                      {player.atpVerified && (
                        <span className="flex items-center gap-1 text-xs text-emerald-600">
                          <ShieldCheck className="h-3 w-3" />
                          Verified
                        </span>
                      )}
                    </div>
                  </CardContent>
                </Card>
              ))}
            </div>
          )}
        </div>
      </main>

      <Footer />
    </div>
  );
}
